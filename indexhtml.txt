        function startTimers() {
            const START_TS = 1715904000000; // May 17, 2024
            setInterval(() => {
                const now = Date.now();
                const diff1 = now - START_TS;
                document.getElementById('days1').innerText = Math.floor(diff1 / 86400000);
                document.getElementById('hours1').innerText = Math.floor((diff1 % 86400000) / 3600000);
                document.getElementById('mins1').innerText = Math.floor((diff1 % 3600000) / 60000);
                document.getElementById('secs1').innerText = Math.floor((diff1 % 60000) / 1000);

                let annivTS = new Date(new Date().getFullYear(), 4, 17).getTime();
                if (now > annivTS) annivTS = new Date(new Date().getFullYear() + 1, 4, 17).getTime();
                const diff2 = annivTS - now;
                document.getElementById('days2').innerText = Math.floor(diff2 / 86400000);
                document.getElementById('hours2').innerText = Math.floor((diff2 % 86400000) / 3600000);
                document.getElementById('mins2').innerText = Math.floor((diff2 % 3600000) / 60000);
                document.getElementById('secs2').innerText = Math.floor((diff2 % 60000) / 1000);
            }, 1000);
        }
        startTimers();

        function navTo(n) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            const target = document.getElementById(`page${n}`);
            if(target) {
                target.classList.add('active');
                if(n >= 1 && n <= 10) type(storyData[n-1].t, `t${n}`, `b${n}`);
                if(n === 11) initWheel();
                if(n === 12) { setTimeout(() => { document.getElementById('fq').style.opacity = 1; runFinale(); }, 500); }
            }
        }

        function type(txt, elId, btnId) {
            const el = document.getElementById(elId);
            if (!el) return;
            el.innerText = "";
            let i = 0;
            const timer = setInterval(() => {
                if(i < txt.length) {
                    el.innerText += txt.charAt(i);
                    el.parentElement.scrollTop = el.parentElement.scrollHeight;
                    i++;
                } else {
                    clearInterval(timer);
                    const btn = document.getElementById(btnId);
                    if (btn) btn.classList.add('visible-btn');
                }
            }, 40);
        }

        // Wheel Logic
        const labels = ["Kiss 💋", "Tight Hug 🫂", "Hand Hold 🤝", "Waist Grab 💖", "Try Again 🔄", "Movie Date 🍿"];
        function initWheel() {
            const can = document.getElementById('wheel'); 
            if (!can) return;
            const ctx = can.getContext('2d');
            const arc = (Math.PI*2)/6;
            labels.forEach((l, i) => {
                ctx.fillStyle = i%2==0 ? "#ffe5ec" : "#ffb3c6";
                ctx.beginPath(); ctx.moveTo(155,155); ctx.arc(155,155,155,i*arc,(i+1)*arc); ctx.fill();
                ctx.save(); ctx.fillStyle="#d00000"; ctx.font="bold 14px Poppins";
                ctx.translate(155+Math.cos(i*arc+arc/2)*105, 155+Math.sin(i*arc+arc/2)*105); ctx.rotate(i*arc+arc/2);
                ctx.fillText(l, -20, 5); ctx.restore();
            });
        }
        function spinWheel() {
            const can = document.getElementById('wheel');
            if (!can) return;
            const deg = 1800 + Math.random()*1800; 
            can.style.transform = `rotate(${deg}deg)`;
            setTimeout(() => { 
                const msg = document.getElementById('spin-msg');
                if (msg) msg.innerText  = "Destiny chose this! ❤️"; 
                const btn = document.getElementById('btn11');
                if (btn) btn.classList.add('visible-btn'); 
            }, 5500);
        }

        function runFinale() {

            if (pup) pup.style.right = "50%";

                const fl = document.getElementById('fl');
                if (fp) fp.classList.add('revealed');
                if (fl) fl.style.display = "block";
            }, 3500);
        }
